CREATE TABLE CLIENTE (
id INT AUTO_INCREMENT NOT NULL,
CPF VARCHAR(11) UNIQUE,
NOME VARCHAR(60),
TELEFONE VARCHAR(20),
CIDADE VARCHAR(20),
EMAIL VARCHAR(50),

CONSTRAINT PK_CliID PRIMARY KEY(id));

CREATE TABLE CORRETOR (
id INT AUTO_INCREMENT NOT NULL,
CRECI VARCHAR(11) UNIQUE,
NOME VARCHAR(60),
TELEFONE VARCHAR(20),
CIDADE VARCHAR(20),
EMAIL VARCHAR(50),

CONSTRAINT PK_CorID PRIMARY KEY(id));

CREATE TABLE IMOVEL (
id INT AUTO_INCREMENT NOT NULL,
Endereco VARCHAR(100) NOT NULL,
Codigo VARCHAR(11) UNIQUE,
AREA DOUBLE(6,2),
VALOR DOUBLE(12,2),
Tipo VARCHAR(20),
QuantComodos INT,

CONSTRAINT PK_ImoID PRIMARY KEY(id));

CREATE TABLE PAGAMENTO (
id INT AUTO_INCREMENT NOT NULL,
IDCLI INT NOT NULL,
IDCOR INT NOT NULL,
IDIMOVEL INT NOT NULL,
DATAPAG DATE,
HORAPAG TIME,

CONSTRAINT PK_PagID PRIMARY KEY(id),
CONSTRAINT FK_IDCLI FOREIGN KEY (IDCLI) REFERENCES CLIENTE(id),
CONSTRAINT FK_IDCOR FOREIGN KEY (IDCOR) REFERENCES CORRETOR(id),
CONSTRAINT FK_IDIMOVEL FOREIGN KEY (IDIMOVEL) REFERENCES IMOVEL(id));
